create database BTL_MMT

use BTL_MMT

CREATE TABLE Users(
	UserName varchar(50) PRIMARY KEY,
	Pass varchar(50) NOT NULL
)

CREATE TABLE Files (
	FileID INT  PRIMARY KEY,
	LocalFileName CHAR(100) ,
	SharedFileName CHAR(100)
);

CREATE TABLE UserFiles(
	UserName varchar(50),
	FileID INT ,
	CONSTRAINT FK_UserFiles_Users FOREIGN KEY (UserName) REFERENCES Users(UserName),
	CONSTRAINT FK_UserFiles_Files FOREIGN KEY (FileID) REFERENCES Files(FileID),
	PRIMARY KEY (UserName, FileID)
)
